46  assert(m_currentTime >= m_contactPhaselist.firstPhase()->beginTime
47  && "[FixedFootDetector::updateFixedFoot] The current time must be greater than "
48  "equal the first contact phase. If you read this assert there is a bug in the code");
49 
50  // search the phase associated to the current time
51  auto phase = std::find_if(m_contactPhaselist.cbegin(),
52  m_contactPhaselist.cend(),
53  [&](const auto& phase) -> bool {
54  return phase.beginTime <= m_currentTime
55  && m_currentTime <= phase.endTime;
56  });
57 
58  // take the last phase if not found
59  if (phase == m_contactPhaselist.cend())
60  phase = m_contactPhaselist.lastPhase();
61 
62  // if the robot is in single support only one contact is enabled
63  if (phase->activeContacts.size() == 1)
64  {
65  // get the enabled contact
66  const auto it = phase->activeContacts.cbegin();
67  assert(m_contactStates.find(it->first) != m_contactStates.end()
68  && "[FixedFootDetector::updateFixedFoot] Unable to find the contact. This "
69  "should not be possible. If you read this assert there is a bug in the code");
70 
71  // update the contacts
72  for (auto& [name, contact] : m_contactStates)
73  {
74  contact.lastUpdateTime = m_currentTime;
75  contact.switchTime = phase->beginTime;
76  if (name != it->first)
77  {
78  contact.isActive = false;
79  } else
80  {
81  contact.isActive = true;
82  contact.pose = it->second->pose;
83  }
84  }
85  } else if (phase->activeContacts.size() == 2)
86  {
87  // Notice that here we do not analyze the case in which the phase is different from
88  // firstPhase. Indeed in this case we should not update the contact states dictionary. (The
89  // content of the dictionary is the one of the previous single support phase)
90 
91  // if the current phase is the first and there are at least 2 phases the active contact will
92  // be the one that is going to be active in the next phase.
93  if (phase == m_contactPhaselist.firstPhase())

96  {

98  const auto it = nextPhase->activeContacts.cbegin();
99

101  && "[FixedFootDetector::updateFixedFoot] Unable to find the contact. "
102  "This should not be possible. If you read this assert there is a bug in "
103  "the code");
104

106  for (auto& [name, contact] : m_contactStates)
107  {
108  contact.lastUpdateTime = m_currentTime;
109  contact.switchTime = phase->beginTime;
110  if (name != it->first)
111  {
112  contact.isActive = true;
113  contact.pose = phase->activeContacts.find(name)->second->pose;
114  } else
115  {
116  contact.isActive = false;
117  }
118  }
119  } else // contactPhaselist.size() == 1
120  {
121  // there is only one contact phase and this phase is a double support phase.
122  // Here we take the first contact as active contact.
123  auto contact = phase->activeContacts.cbegin();
124

126  && "[FixedFootDetector::updateFixedFoot] Unable to find the contact. "

131  m_contactStates[contact->first].switchTime = phase->beginTime;
132  m_contactStates[contact->first].isActive = true;
133  m_contactStates[contact->first].pose = contact->second->pose;
134

145  }
146  }
